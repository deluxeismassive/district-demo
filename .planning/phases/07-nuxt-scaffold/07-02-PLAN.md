== Canonical app/stores/tags.ts (copy verbatim) ==

```ts
import { defineStore } from 'pinia'
import { ref } from 'vue'

export interface TagItem {
  id: string
  name: string
}

export interface TagGroup {
  id: string
  name: string
  color: string
  children: TagItem[]
}

export type TagAssignments = Record<string, string[]>

export const SEED_TAG_GROUPS: TagGroup[] = [
  {
    id: 'group-curriculum',
    name: 'Curriculum',
    color: '#484ce6',
    children: [
      { id: 'tag-math', name: 'Math' },
      { id: 'tag-science', name: 'Science' },
      { id: 'tag-ela', name: 'ELA' },
      { id: 'tag-social-studies', name: 'Social Studies' },
    ],
  },
  {
    id: 'group-assessment',
    name: 'Assessment',
    color: '#da8231',
    children: [
      { id: 'tag-formative', name: 'Formative' },
      { id: 'tag-summative', name: 'Summative' },
    ],
  },
  {
    id: 'group-communication',
    name: 'Communication',
    color: '#16a34a',
    children: [
      { id: 'tag-parent-engagement', name: 'Parent Engagement' },
      { id: 'tag-staff-messaging', name: 'Staff Messaging' },
    ],
  },
  {
    id: 'group-admin',
    name: 'Administration',
    color: '#dc2626',
    children: [
      { id: 'tag-hr', name: 'HR' },
      { id: 'tag-finance', name: 'Finance' },
      { id: 'tag-scheduling', name: 'Scheduling' },
    ],
  },
]

export const useTagsStore = defineStore('tags', () => {
  const tagGroups = ref<TagGroup[]>(SEED_TAG_GROUPS)
  const assignments = ref<TagAssignments>({})

  return { tagGroups, assignments }
}, {
  persist: true,
})
```

== Pitfalls (must avoid — from research Sections 4, 5 + Pitfalls 2, 4, 6) ==

1. DO NOT call `localStorage.getItem()` in store setup — runs on server, crashes SSR.
2. DO NOT use `process.client` guards — the plugin handles client-only hydration automatically.
3. DO NOT wrap `<VChart>` in `<ClientOnly>` — `nuxt-echarts` handles SSR via SVG fallback.
4. DO NOT reorder modules: `'@pinia/nuxt'` must come BEFORE `'pinia-plugin-persistedstate/nuxt'`.
5. DO NOT set per-store `persist: { storage: localStorage }` — global `storage: 'localStorage'` in `nuxt.config.ts` already covers it; per-store config would access `localStorage` at setup time and crash SSR. Use `persist: true` (boolean) only.
